<div id="list"></div>

<script>
let sentences = [];
let index = 0;
let cache = {};

// 分句
function build(){
  let text = document.getElementById("input").value;

  sentences = text.split(/\n+/).filter(s => s.trim());

  let html = "";

  sentences.forEach((s,i)=>{
    html += `
    <div class="item" id="item${i}" onclick="play(${i})">
      <div class="ar">${s}</div>
      <div class="zh" id="zh${i}">翻译加载中...</div>
    </div>`;
  });

  document.getElementById("list").innerHTML = html;

  sentences.forEach((s,i)=> translate(s,i));
}

// 🔊 语音
function play(i){
  highlight(i);

  let u = new SpeechSynthesisUtterance(sentences[i]);
  u.lang = "ar-SA";

  speechSynthesis.cancel();
  speechSynthesis.speak(u);

  index = i;
}

// 高亮
function highlight(i){
  document.querySelectorAll(".item").forEach(e=>e.style.background="white");
  document.getElementById("item"+i).style.background="#cde6ff";
}

// 连播
function playAll(){
  index = 0;
  next();
}

function next(){
  if(index >= sentences.length) return;

  play(index);

  setTimeout(()=>{
    index++;
    next();
  }, 2500);
}

// 停止
function stop(){
  speechSynthesis.cancel();
}

// 🌐 翻译（核心升级）
async function translate(text,i){

  if(cache[text]){
    document.getElementById("zh"+i).innerText = cache[text];
    return;
  }

  try {
    let res = await fetch(
      "https://api.mymemory.translated.net/get?q="
      + encodeURIComponent(text)
      + "&langpair=ar|zh-CN"
    );

    let data = await res.json();

    let zh = data.responseData.translatedText;

    cache[text] = zh;

    document.getElementById("zh"+i).innerText = zh;

  } catch(e){
    document.getElementById("zh"+i).innerText = "翻译失败";
  }
}
</script>
